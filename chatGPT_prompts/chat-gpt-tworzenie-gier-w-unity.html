import pygame

# Initialize Pygame
pygame.init()

# Set up the screen
screen_width = 800
screen_height = 600
screen = pygame.display.set_mode((screen_width, screen_height))
pygame.display.set_caption("Frog Game")

# Set up the frog object
frog_width = 50
frog_height = 50
frog_x = screen_width // 2 - frog_width // 2
frog_y = screen_height // 2 - frog_height // 2
frog_speed = 5
frog = pygame.Rect(frog_x, frog_y, frog_width, frog_height)

# Set up the boundaries
boundary_top = pygame.Rect(0, 0, screen_width, 50)
boundary_bottom = pygame.Rect(0, screen_height - 50, screen_width, 50)
boundary_left = pygame.Rect(0, 0, 50, screen_height)
boundary_right = pygame.Rect(screen_width - 50, 0, 50, screen_height)

# Set up the clock
clock = pygame.time.Clock()

# Main game loop
running = True
while running:
    # Handle events
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # Move the frog object using arrow keys
    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT] and frog.left > boundary_left.right:
        frog.x -= frog_speed
    if keys[pygame.K_RIGHT] and frog.right < boundary_right.left:
        frog.x += frog_speed
    if keys[pygame.K_UP] and frog.top > boundary_top.bottom:
        frog.y -= frog_speed
    if keys[pygame.K_DOWN] and frog.bottom < boundary_bottom.top:
        frog.y += frog_speed

    # Draw the game objects
    screen.fill((255, 255, 255))  # Clear the screen
    pygame.draw.rect(screen, (0, 255, 0), frog)  # Draw the frog object
    pygame.draw.rect(screen, (255, 0, 0), boundary_top)  # Draw the top boundary
    pygame.draw.rect(screen, (255, 0, 0), boundary_bottom)  # Draw the bottom boundary
    pygame.draw.rect(screen, (255, 0, 0), boundary_left)  # Draw the left boundary
    pygame.draw.rect(screen, (255, 0, 0), boundary_right)  # Draw the right boundary

    # Update the screen
    pygame.display.flip()

    # Limit the frame rate to 60 FPS
    clock.tick(60)

# Clean up Pygame
pygame.quit()

This code sets up a frog object that can be moved using arrow keys, and sets up four boundary rectangles to keep the frog within the screen. The game loop handles user input, updates the frog object's position, and draws the game objects on the screen. The clock is used to limit the frame rate to 60 FPS.
